cmake_minimum_required(VERSION 3.16)
project(fast_agent_ingest_cpp CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# ── Native library ────────────────────────────────────────────────────────────
# The Rust C ABI library must be built first:
#   cargo build --package fast-agent-ingest-c-ffi --release

set(FAI_LIB_DIR "${CMAKE_CURRENT_SOURCE_DIR}/../../target/release"
    CACHE PATH "Directory containing libfast_agent_ingest_c_ffi")

find_library(FAI_LIBRARY
    NAMES fast_agent_ingest_c_ffi
    PATHS "${FAI_LIB_DIR}"
    REQUIRED)

# ── Header ────────────────────────────────────────────────────────────────────
# The C header is generated by cbindgen (just generate-headers).
set(FAI_INCLUDE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/../../include")

add_library(fast_agent_ingest INTERFACE)
target_include_directories(fast_agent_ingest INTERFACE
    "${CMAKE_CURRENT_SOURCE_DIR}/include"
    "${FAI_INCLUDE_DIR}")
target_link_libraries(fast_agent_ingest INTERFACE "${FAI_LIBRARY}")

# On Linux, also link against dl and pthread (required by Rust stdlib).
if(UNIX AND NOT APPLE)
    target_link_libraries(fast_agent_ingest INTERFACE dl pthread)
endif()

# ── Tests ─────────────────────────────────────────────────────────────────────
enable_testing()
add_subdirectory(tests)
